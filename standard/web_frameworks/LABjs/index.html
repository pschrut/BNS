LABjs (Loading And Blocking JavaScript)

NOTE: This is a live demo test script for LABjs (Loading And Blocking JavaScript). Read more about this project in the Project Description.

Current version: 0.5.1 LABjs-0.5.1.zip.

Questions or Thoughts? Contact me @getify on twitter.

If you like this project, you also might like to check out mpAjax (multi-part Ajax responses) and flXHR (easy cross-domain Ajax). 

Example 1:

	$LAB
	.script("script1.js")
	.script("script2.js")
	.script("script3.js")
	.block(function(){
		// wait for all to load, then do something
		script1Func();
		script2Func();
		script3Func();
	});


Example 2:

	$LAB	
	.script({ src: "script1.js", type: "text/javascript" })
	.script({ src: "script2.js", language: "javascript", type: "text/javascript" })
	.script("script3.js")
	.block(function(){
		// wait for all to load, then do something
		script1Func();
		script2Func();
		script3Func();
	});


Example 3:

	$LAB
	.script("script1.js", "script2.js", "script3.js")
	.block(function(){
		// wait for all to load, then do something
		script1Func();
		script2Func();
		script3Func();
	});


Example 4:
	
	$LAB
	.script( [ "script1.js", "script2.js" ], "script3.js")
	.block(function(){
		// wait for all to load, then do something
		script1Func();
		script2Func();
		script3Func();
	});
	
	
Example 5:

	$LAB
	.script("script1.js")
	.block(function(){
		$LAB
		.script("script2.js") // depends on script1 being loaded before
		.script("script3.js") // also depends on script1 being loaded before
		.block(function(){
			$LAB
			.script("script4.js") // depends on script1, script2 and script3 being loaded before
			.block(function(){
				script4Func();
			});
		});
	});

Example 6:

	$LAB
	.script("script1.js") // defaults to appending to the HEAD
	.block(function(){
		$LAB
		.toBODY() // will append everything else to the BODY
		.script("script2.js") // depends on script1 being loaded before
		.script("script3.js") // also depends on script1 being loaded before
		.block(function(){
			$LAB
			.toHEAD() // switch back to HEAD appending if you want
			.script("script4.js") // depends on script1, script2 and script3 being loaded before
			.block(function(){
				script4Func();
			});
		});
	});
