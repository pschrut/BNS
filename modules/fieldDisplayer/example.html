Field Displayer example

Two input fields, mandatory with automatic format check

Depent fields

Code

        <div id="fields"></div>
        <script type="text/javascript">
            var fieldDisplayer = new FieldDisplayer({fieldFormat: 'I', fieldId: 'FIELD1', displayAttrib: 'MAN', fieldLabel: 'Insert a number', type: 'NUMC'});
            var fieldDisplayerText= new FieldDisplayer({fieldFormat: 'I', fieldId: 'FIELD1', displayAttrib: 'MAN', fieldLabel: 'Insert your name', type: 'CHAR'});
            $('fields').insert(fieldDisplayer.getElement());
            $('fields').insert(fieldDisplayerText.getElement());
        </script>
